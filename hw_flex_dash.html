Rating proportions by boro

bar_plot = nyc_airbnb %>% 
  mutate(rating = as.character(rating)) %>%
  group_by(boro, rating) %>%
  summarize(count = n()) %>%
  group_by(boro) %>% 
  mutate(sum = sum(count)) %>% 
  mutate(proportion = count/sum) %>%
  ggplot(aes(x = boro, y = proportion, fill = rating)) + 
  geom_bar(width = 0.9, position = position_dodge(width=1.0), stat="identity") +
  theme_bw()
